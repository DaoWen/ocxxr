42  inline void *allocateAligned(size_t size, int alignment) const {
43  assert(m_info != nullptr && "Uninitialized allocator");
44  const ptrdiff_t offset = m_info->offset;
45  ptrdiff_t start = (offset + alignment - 1) & (-alignment);
46  m_info->offset = start + size;
47  assert(m_info->offset <= m_info->size &&
48  "Datablock allocator overflow");
49  return &m_dbBuf[start];
50  }
